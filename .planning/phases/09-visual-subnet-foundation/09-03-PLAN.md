    Create Zustand store following project pattern (see context: calculator-store.ts).

    **Imports:**
    ```typescript
    "use client";

    import { create } from 'zustand';
    import { createUrlSyncMiddleware } from '@/lib/middleware/url-sync';
    import { getUrlParams, parseStringParam } from '@/lib/utils/url-params';
    import { calculateSubnet } from '@/lib/converters/network/subnet-calculator';
    import { parseIPInput } from '@/lib/converters/network/ip-parser';
    import type { SubnetResult } from '@/lib/converters/network/types';
    ```

    **State interface:**
    ```typescript
    export interface SubnetCalculatorState {
      // Inputs
      ipInput: string;
      subnetMask: string;

      // Results
      result: SubnetResult | null;
      error: string | null;

      // Actions
      setIPInput: (value: string) => void;
      setSubnetMask: (value: string) => void;
      calculate: () => void;
      reset: () => void;
    }
    ```

    **Initial state:**
    ```typescript
    const initialState = {
      ipInput: '',
      subnetMask: '',
      result: null,
      error: null,
    };
    ```

    **Store implementation:**
    - Use createUrlSyncMiddleware with selectState: (state) => ({ ipInput: state.ipInput, subnetMask: state.subnetMask })
    - debounceMs: 300
    - Load URL params on mount using getUrlParams() and parseStringParam()

    **setIPInput action:**
    - Update ipInput
    - Clear error
    - Auto-calculate if input contains "/" (CIDR notation complete)

    **setSubnetMask action:**
    - Update subnetMask
    - Clear error
    - Auto-calculate if both ipInput and subnetMask present

    **calculate action:**
    - Try-catch block around parseIPInput() and calculateSubnet()
    - On success: set result, clear error
    - On error: set error message, clear result
    - Error messages from parsing/calculation are user-friendly

    **reset action:**
    - Reset to initialState

    Follow RESEARCH.md Pattern 3 for store structure.

    **page.tsx (server component):**

    ```typescript
    import { getTranslations, setRequestLocale } from "next-intl/server";
    import type { Metadata } from "next";
    import { locales } from "@/i18n/config";
    import { ConverterLayout } from "@/components/converter/converter-layout";
    import { SubnetCalculator } from "./subnet-calculator";

    export async function generateStaticParams() {
      return locales.map((locale) => ({ locale }));
    }

    export async function generateMetadata({
      params,
    }: {
      params: Promise<{ locale: string }>;
    }): Promise<Metadata> {
      const { locale } = await params;
      const t = await getTranslations({ locale, namespace: "converters.subnet-calculator" });

      return {
        title: t("name"),
        description: t("metaDescription"),
      };
    }

    export default async function SubnetCalculatorPage({
      params,
    }: {
      params: Promise<{ locale: string }>;
    }) {
      const { locale } = await params;
      setRequestLocale(locale);
      const t = await getTranslations("converters.subnet-calculator");

      return (
        <ConverterLayout
          title={t("name")}
          description={t("description")}
        >
          <SubnetCalculator />
        </ConverterLayout>
      );
    }
    ```
