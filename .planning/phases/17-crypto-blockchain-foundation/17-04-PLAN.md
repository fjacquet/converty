#!/usr/bin/env npx ts-node
/**
 * Fetch Bitcoin mining network data at build time
 * Run with: npx ts-node scripts/fetch-mining-data.ts
 */

import * as fs from "fs";
import * as path from "path";

interface MiningData {
  bitcoin: {
    difficulty: number;
    networkHashRate: number;  // TH/s
    blockReward: number;      // BTC
    blocksPerDay: number;
    currentPrice: {
      chf: number;
      eur: number;
      usd: number;
    };
  };
  lastUpdated: string;
  source: string;
}

const BLOCKCHAIN_API = "https://blockchain.info/q";
const COINGECKO_API = "https://api.coingecko.com/api/v3/simple/price";

async function fetchDifficulty(): Promise<number> {
  try {
    const response = await fetch(`${BLOCKCHAIN_API}/getdifficulty`);
    if (!response.ok) throw new Error("Failed to fetch difficulty");
    const text = await response.text();
    return Number.parseFloat(text);
  } catch {
    // Fallback difficulty (approximate Dec 2025 value)
    return 100000000000000; // 100 trillion
  }
}

async function fetchHashRate(): Promise<number> {
  try {
    const response = await fetch(`${BLOCKCHAIN_API}/hashrate`);
    if (!response.ok) throw new Error("Failed to fetch hash rate");
    const text = await response.text();
    // API returns in H/s, convert to TH/s
    return Number.parseFloat(text) / 1e12;
  } catch {
    // Fallback hash rate (approximate Dec 2025 value)
    return 500000000; // 500 EH/s in TH/s
  }
}

async function fetchBitcoinPrice(): Promise<{ chf: number; eur: number; usd: number }> {
  try {
    const url = `${COINGECKO_API}?ids=bitcoin&vs_currencies=chf,eur,usd`;
    const response = await fetch(url);
    if (!response.ok) throw new Error("Failed to fetch price");
    const data = await response.json();
    return data.bitcoin;
  } catch {
    // Fallback prices
    return { chf: 85000, eur: 90000, usd: 95000 };
  }
}

async function main() {
  try {
    console.log("Fetching Bitcoin mining data...");

    const [difficulty, networkHashRate, currentPrice] = await Promise.all([
      fetchDifficulty(),
      fetchHashRate(),
      fetchBitcoinPrice(),
    ]);

    // Current block reward (post-2024 halving)
    const blockReward = 3.125;

    // Blocks per day (144 average)
    const blocksPerDay = 144;

    const data: MiningData = {
      bitcoin: {
        difficulty,
        networkHashRate,
        blockReward,
        blocksPerDay,
        currentPrice,
      },
      lastUpdated: new Date().toISOString(),
      source: "Blockchain.info + CoinGecko API",
    };

    const outputPath = path.join(process.cwd(), "src/lib/data/mining-data.json");

    // Ensure directory exists
    const dir = path.dirname(outputPath);
    if (!fs.existsSync(dir)) {
      fs.mkdirSync(dir, { recursive: true });
    }

    fs.writeFileSync(outputPath, JSON.stringify(data, null, 2));

    console.log(`Mining data saved to ${outputPath}`);
    console.log("Data:", JSON.stringify(data, null, 2));
  } catch (error) {
    console.error("Failed to fetch mining data:", error);
    createFallbackData();
  }
}

function createFallbackData() {
  console.log("Using fallback mining data...");

  const fallbackData: MiningData = {
    bitcoin: {
      difficulty: 100000000000000,
      networkHashRate: 500000000,
      blockReward: 3.125,
      blocksPerDay: 144,
      currentPrice: {
        chf: 85000,
        eur: 90000,
        usd: 95000,
      },
    },
    lastUpdated: new Date().toISOString(),
    source: "Fallback (API unavailable)",
  };

  const outputPath = path.join(process.cwd(), "src/lib/data/mining-data.json");
  const dir = path.dirname(outputPath);

  if (!fs.existsSync(dir)) {
    fs.mkdirSync(dir, { recursive: true });
  }

  fs.writeFileSync(outputPath, JSON.stringify(fallbackData, null, 2));
  console.log(`Fallback mining data saved to ${outputPath}`);
}

main();

import miningData from "@/lib/data/mining-data.json";

export type HashRateUnit = "H/s" | "KH/s" | "MH/s" | "GH/s" | "TH/s" | "PH/s";
export type FiatCurrency = "CHF" | "EUR" | "USD";

export interface MiningInput {
  hashRate: number;
  hashRateUnit: HashRateUnit;
  powerWatts: number;
  electricityCost: number;   // Per kWh
  currency: FiatCurrency;
  hardwareCost?: number;     // Optional for ROI calculation
}

export interface MiningResult {
  // Revenue
  revenuePerDay: number;
  revenuePerMonth: number;
  revenuePerYear: number;

  // Electricity cost
  electricityCostPerDay: number;
  electricityCostPerMonth: number;
  electricityCostPerYear: number;

  // Profit
  profitPerDay: number;
  profitPerMonth: number;
  profitPerYear: number;

  // Mining output
  btcPerDay: number;
  btcPerMonth: number;
  btcPerYear: number;

  // ROI (if hardware cost provided)
  roiDays: number | null;
  roiMonths: number | null;
  breakEvenDate: string | null;

  // Context
  isProfitable: boolean;
  currency: FiatCurrency;
  btcPrice: number;
  networkDifficulty: number;
  blockReward: number;
  lastUpdated: string;
  source: string;
}

/**
 * Hash rate unit multipliers to convert to H/s
 */
const HASH_RATE_MULTIPLIERS: Record<HashRateUnit, number> = {
  "H/s": 1,
  "KH/s": 1e3,
  "MH/s": 1e6,
  "GH/s": 1e9,
  "TH/s": 1e12,
  "PH/s": 1e15,
};

/**
 * Available hash rate units for UI
 */
export const HASH_RATE_UNITS: { unit: HashRateUnit; name: string }[] = [
  { unit: "H/s", name: "H/s (Hash/second)" },
  { unit: "KH/s", name: "KH/s (Kilohash/second)" },
  { unit: "MH/s", name: "MH/s (Megahash/second)" },
  { unit: "GH/s", name: "GH/s (Gigahash/second)" },
  { unit: "TH/s", name: "TH/s (Terahash/second)" },
  { unit: "PH/s", name: "PH/s (Petahash/second)" },
];

/**
 * Get BTC price in specified currency
 */
function getBtcPrice(currency: FiatCurrency): number {
  const prices = miningData.bitcoin.currentPrice;
  switch (currency) {
    case "CHF":
      return prices.chf;
    case "EUR":
      return prices.eur;
    case "USD":
      return prices.usd;
    default:
      return prices.chf;
  }
}

/**
 * Calculate mining profitability
 *
 * Formula:
 * BTC per day = (HashRate / NetworkHashRate) * BlocksPerDay * BlockReward
 * Revenue = BTC per day * BTC Price
 * Electricity Cost = (PowerWatts / 1000) * 24 * ElectricityCostPerKWh
 * Profit = Revenue - Electricity Cost
 */
export function calculateMiningProfitability(input: MiningInput): MiningResult {
  const { hashRate, hashRateUnit, powerWatts, electricityCost, currency, hardwareCost } = input;

  // Convert hash rate to TH/s
  const hashRateInHs = hashRate * HASH_RATE_MULTIPLIERS[hashRateUnit];
  const hashRateInTHs = hashRateInHs / 1e12;

  // Get network data
  const networkHashRateTHs = miningData.bitcoin.networkHashRate;
  const blocksPerDay = miningData.bitcoin.blocksPerDay;
  const blockReward = miningData.bitcoin.blockReward;
  const btcPrice = getBtcPrice(currency);

  // Calculate BTC mined per day
  // Share of network hash rate * blocks per day * block reward
  const btcPerDay = (hashRateInTHs / networkHashRateTHs) * blocksPerDay * blockReward;
  const btcPerMonth = btcPerDay * 30;
  const btcPerYear = btcPerDay * 365;

  // Calculate revenue
  const revenuePerDay = btcPerDay * btcPrice;
  const revenuePerMonth = btcPerMonth * btcPrice;
  const revenuePerYear = btcPerYear * btcPrice;

  // Calculate electricity cost
  const kWhPerDay = (powerWatts / 1000) * 24;
  const electricityCostPerDay = kWhPerDay * electricityCost;
  const electricityCostPerMonth = electricityCostPerDay * 30;
  const electricityCostPerYear = electricityCostPerDay * 365;

  // Calculate profit
  const profitPerDay = revenuePerDay - electricityCostPerDay;
  const profitPerMonth = revenuePerMonth - electricityCostPerMonth;
  const profitPerYear = revenuePerYear - electricityCostPerYear;

  // Calculate ROI (if hardware cost provided and profitable)
  let roiDays: number | null = null;
  let roiMonths: number | null = null;
  let breakEvenDate: string | null = null;

  if (hardwareCost && hardwareCost > 0 && profitPerDay > 0) {
    roiDays = Math.ceil(hardwareCost / profitPerDay);
    roiMonths = roiDays / 30;

    const breakEven = new Date();
    breakEven.setDate(breakEven.getDate() + roiDays);
    breakEvenDate = breakEven.toISOString().split("T")[0];
  }

  return {
    revenuePerDay,
    revenuePerMonth,
    revenuePerYear,
    electricityCostPerDay,
    electricityCostPerMonth,
    electricityCostPerYear,
    profitPerDay,
    profitPerMonth,
    profitPerYear,
    btcPerDay,
    btcPerMonth,
    btcPerYear,
    roiDays,
    roiMonths,
    breakEvenDate,
    isProfitable: profitPerDay > 0,
    currency,
    btcPrice,
    networkDifficulty: miningData.bitcoin.difficulty,
    blockReward,
    lastUpdated: miningData.lastUpdated,
    source: miningData.source,
  };
}

/**
 * Format currency amount
 */
export function formatMiningCurrency(amount: number, currency: FiatCurrency): string {
  return amount.toLocaleString(undefined, {
    style: "currency",
    currency,
    minimumFractionDigits: 2,
    maximumFractionDigits: 2,
  });
}

/**
 * Format BTC amount
 */
export function formatBtc(amount: number): string {
  if (amount >= 1) {
    return amount.toFixed(6);
  }
  if (amount >= 0.00001) {
    return amount.toFixed(8);
  }
  return amount.toExponential(4);
}

/**
 * Get data age in hours
 */
export function getMiningDataAge(): number {
  const lastUpdated = new Date(miningData.lastUpdated);
  const now = new Date();
  const diffMs = now.getTime() - lastUpdated.getTime();
  return Math.floor(diffMs / (1000 * 60 * 60));
}

/**

export function getMiningLastUpdated(): string {
  return miningData.lastUpdated;
}

// src/stores/mining-calculator-store.ts
"use client";

import { create } from "zustand";
import {
  calculateMiningProfitability,
  type HashRateUnit,
  type FiatCurrency,
  type MiningResult,
} from "@/lib/converters/crypto/mining-profitability";
import { createUrlSyncMiddleware } from "@/lib/middleware/url-sync";
import { getUrlParams, parseStringParam, parseNumberParam } from "@/lib/utils/url-params";

export interface MiningCalculatorState {
  // Input
  hashRate: string;
  hashRateUnit: HashRateUnit;
  powerWatts: string;
  electricityCost: string;
  currency: FiatCurrency;
  hardwareCost: string;

  // Result
  result: MiningResult | null;
  error: string | null;

  // Actions
  setHashRate: (value: string) => void;
  setHashRateUnit: (unit: HashRateUnit) => void;
  setPowerWatts: (value: string) => void;
  setElectricityCost: (value: string) => void;
  setCurrency: (currency: FiatCurrency) => void;
  setHardwareCost: (value: string) => void;
  applyPreset: (hashRate: number, hashRateUnit: HashRateUnit, powerWatts: number) => void;
  calculate: () => void;
  reset: () => void;
}

const initialState = {
  hashRate: "100",
  hashRateUnit: "TH/s" as HashRateUnit,
  powerWatts: "3000",
  electricityCost: "0.27",  // Swiss residential average
  currency: "CHF" as FiatCurrency,
  hardwareCost: "",
  result: null,
  error: null,
};

export const useMiningCalculatorStore = create<MiningCalculatorState>()(
  createUrlSyncMiddleware<MiningCalculatorState>({
    enabled: true,
    debounceMs: 300,
    selectState: (state) => ({
      hashRate: state.hashRate,
      hashRateUnit: state.hashRateUnit,
      powerWatts: state.powerWatts,
      electricityCost: state.electricityCost,
      currency: state.currency,
      hardwareCost: state.hardwareCost,
    }),
  })((set, get) => {
    // Load initial values from URL params
    let loaded = { ...initialState };

    if (typeof window !== "undefined") {
      const urlParams = getUrlParams();
      if (Object.keys(urlParams).length > 0) {
        loaded.hashRate = parseStringParam(urlParams.hashRate, initialState.hashRate);
        loaded.powerWatts = parseStringParam(urlParams.powerWatts, initialState.powerWatts);
        loaded.electricityCost = parseStringParam(urlParams.electricityCost, initialState.electricityCost);
        loaded.hardwareCost = parseStringParam(urlParams.hardwareCost, initialState.hardwareCost);

        const unitParam = parseStringParam(urlParams.hashRateUnit, initialState.hashRateUnit);
        if (["H/s", "KH/s", "MH/s", "GH/s", "TH/s", "PH/s"].includes(unitParam)) {
          loaded.hashRateUnit = unitParam as HashRateUnit;
        }

        const currencyParam = parseStringParam(urlParams.currency, initialState.currency);
        if (["CHF", "EUR", "USD"].includes(currencyParam)) {
          loaded.currency = currencyParam as FiatCurrency;
        }
      }
    }

    return {
      ...loaded,
      result: null,
      error: null,

      setHashRate: (value: string) => {
        set({ hashRate: value, error: null });
        setTimeout(() => get().calculate(), 0);
      },

      setHashRateUnit: (hashRateUnit: HashRateUnit) => {
        set({ hashRateUnit, error: null });
        setTimeout(() => get().calculate(), 0);
      },

      setPowerWatts: (value: string) => {
        set({ powerWatts: value, error: null });
        setTimeout(() => get().calculate(), 0);
      },

      setElectricityCost: (value: string) => {
        set({ electricityCost: value, error: null });
        setTimeout(() => get().calculate(), 0);
      },

      setCurrency: (currency: FiatCurrency) => {
        set({ currency, error: null });
        setTimeout(() => get().calculate(), 0);
      },

      setHardwareCost: (value: string) => {
        set({ hardwareCost: value, error: null });
        setTimeout(() => get().calculate(), 0);
      },

      applyPreset: (hashRate: number, hashRateUnit: HashRateUnit, powerWatts: number) => {
        set({
          hashRate: hashRate.toString(),
          hashRateUnit,
          powerWatts: powerWatts.toString(),
          error: null,
        });
        setTimeout(() => get().calculate(), 0);
      },

      calculate: () => {
        const { hashRate, hashRateUnit, powerWatts, electricityCost, currency, hardwareCost } = get();

        const numHashRate = Number.parseFloat(hashRate);
        const numPowerWatts = Number.parseFloat(powerWatts);
        const numElectricityCost = Number.parseFloat(electricityCost);
        const numHardwareCost = hardwareCost ? Number.parseFloat(hardwareCost) : undefined;

        if (Number.isNaN(numHashRate) || numHashRate <= 0) {
          set({ result: null, error: null });
          return;
        }

        if (Number.isNaN(numPowerWatts) || numPowerWatts < 0) {
          set({ result: null, error: null });
          return;
        }

        if (Number.isNaN(numElectricityCost) || numElectricityCost < 0) {
          set({ result: null, error: null });
          return;
        }

        try {
          const result = calculateMiningProfitability({
            hashRate: numHashRate,
            hashRateUnit,
            powerWatts: numPowerWatts,
            electricityCost: numElectricityCost,
            currency,
            hardwareCost: numHardwareCost,
          });
          set({ result, error: null });
        } catch (err) {
          set({
            result: null,
            error: err instanceof Error ? err.message : "Calculation failed",
          });
        }
      },

      reset: () => set({ ...initialState, result: null }),
    };
  })
);

import { useTranslations } from "next-intl";
import { Alert, AlertDescription, AlertTitle } from "@/components/ui/alert";
import { Button } from "@/components/ui/button";
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";
import { Input } from "@/components/ui/input";
import { Label } from "@/components/ui/label";
import {
  Select,
  SelectContent,
  SelectItem,
  SelectTrigger,
  SelectValue,
} from "@/components/ui/select";
import {
  HASH_RATE_UNITS,
  MINER_PRESETS,
  formatBtc,
  formatMiningCurrency,
  getMiningDataAge,
  getMiningLastUpdated,
  type HashRateUnit,
  type FiatCurrency,
} from "@/lib/converters/crypto/mining-profitability";
import { useMiningCalculatorStore } from "@/stores/mining-calculator-store";

const CURRENCIES: { id: FiatCurrency; name: string }[] = [
  { id: "CHF", name: "Swiss Franc (CHF)" },
  { id: "EUR", name: "Euro (EUR)" },
  { id: "USD", name: "US Dollar (USD)" },
];

export function MiningCalculator() {
  const t = useTranslations("calculator.crypto.mining");
  const commonT = useTranslations("common");
  const {
    hashRate,
    hashRateUnit,
    powerWatts,
    electricityCost,
    currency,
    hardwareCost,
    result,
    error,
    setHashRate,
    setHashRateUnit,
    setPowerWatts,
    setElectricityCost,
    setCurrency,
    setHardwareCost,
    applyPreset,
    reset,
  } = useMiningCalculatorStore();

  const dataAge = getMiningDataAge();
  const lastUpdated = getMiningLastUpdated();

  return (
    <div className="space-y-6">
      {/* Educational Disclaimer */}
      <Alert>
        <AlertTriangle className="h-4 w-4" />
        <AlertTitle>{t("disclaimer.title")}</AlertTitle>
        <AlertDescription>{t("disclaimer.description")}</AlertDescription>
      </Alert>

      {/* Miner Presets */}
      <Card>
        <CardHeader>
          <CardTitle className="flex items-center gap-2">
            <Cpu className="h-5 w-5" />
            {t("minerPresets")}
          </CardTitle>
        </CardHeader>
        <CardContent>
          <div className="flex flex-wrap gap-2">
            {MINER_PRESETS.filter((p) => p.name !== "Custom").map((preset) => (
              <Button
                key={preset.name}
                variant="outline"
                size="sm"
                onClick={() => applyPreset(preset.hashRate, preset.hashRateUnit, preset.powerWatts)}
              >
                {preset.name}
              </Button>
            ))}
          </div>
        </CardContent>
      </Card>

      {/* Input Section */}
      <Card>
        <CardHeader>
          <CardTitle className="flex items-center gap-2">
            <Calculator className="h-5 w-5" />
            {t("miningParameters")}
          </CardTitle>
        </CardHeader>
        <CardContent className="space-y-4">
          {/* Hash Rate */}
          <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
            <div className="space-y-2">
              <Label htmlFor="hashRate">{t("hashRate")}</Label>
              <Input
                id="hashRate"
                type="number"
                value={hashRate}
                onChange={(e) => setHashRate(e.target.value)}
                placeholder="100"
                min="0"
                step="any"
              />
            </div>
            <div className="space-y-2">
              <Label htmlFor="hashRateUnit">{t("hashRateUnit")}</Label>
              <Select value={hashRateUnit} onValueChange={(v) => setHashRateUnit(v as HashRateUnit)}>
                <SelectTrigger id="hashRateUnit">
                  <SelectValue />
                </SelectTrigger>
                <SelectContent>
                  {HASH_RATE_UNITS.map((u) => (
                    <SelectItem key={u.unit} value={u.unit}>
                      {u.name}
                    </SelectItem>
                  ))}
                </SelectContent>
              </Select>
            </div>
          </div>

          {/* Power and Electricity */}
          <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
            <div className="space-y-2">
              <Label htmlFor="powerWatts">
                <Zap className="inline h-4 w-4 mr-1" />
                {t("powerConsumption")} (W)
              </Label>
              <Input
                id="powerWatts"
                type="number"
                value={powerWatts}
                onChange={(e) => setPowerWatts(e.target.value)}
                placeholder="3000"
                min="0"
              />
            </div>
            <div className="space-y-2">
              <Label htmlFor="electricityCost">{t("electricityCost")} (per kWh)</Label>
              <Input
                id="electricityCost"
                type="number"
                value={electricityCost}
                onChange={(e) => setElectricityCost(e.target.value)}
                placeholder="0.27"
                min="0"
                step="0.01"
              />
            </div>
          </div>

          {/* Currency and Hardware Cost */}
          <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
            <div className="space-y-2">
              <Label htmlFor="currency">{t("currency")}</Label>
              <Select value={currency} onValueChange={(v) => setCurrency(v as FiatCurrency)}>
                <SelectTrigger id="currency">
                  <SelectValue />
                </SelectTrigger>
                <SelectContent>
                  {CURRENCIES.map((c) => (
                    <SelectItem key={c.id} value={c.id}>
                      {c.name}
                    </SelectItem>
                  ))}
                </SelectContent>
              </Select>
            </div>
            <div className="space-y-2">
              <Label htmlFor="hardwareCost">{t("hardwareCost")} ({t("optional")})</Label>
              <Input
                id="hardwareCost"
                type="number"
                value={hardwareCost}
                onChange={(e) => setHardwareCost(e.target.value)}
                placeholder="5000"
                min="0"
              />
            </div>
          </div>
        </CardContent>
      </Card>

      {/* Error Display */}
      {error && (
        <Card className="border-destructive">
          <CardContent className="pt-6">
            <p className="text-destructive">{error}</p>
          </CardContent>
        </Card>
      )}

      {/* Results Section */}
      {result && (
        <>
          {/* Profitability Summary */}
          <Card className={result.isProfitable ? "border-green-500" : "border-destructive"}>
            <CardHeader>
              <CardTitle className="flex items-center gap-2">
                {result.isProfitable ? (
                  <TrendingUp className="h-5 w-5 text-green-500" />
                ) : (
                  <TrendingDown className="h-5 w-5 text-destructive" />
                )}
                {result.isProfitable ? t("profitable") : t("notProfitable")}
              </CardTitle>
            </CardHeader>
            <CardContent>
              <div className="text-center p-4 bg-muted rounded-lg">
                <div className="text-sm text-muted-foreground mb-1">{t("dailyProfit")}</div>
                <div className={`text-3xl font-bold ${result.isProfitable ? "text-green-600" : "text-destructive"}`}>
                  {formatMiningCurrency(result.profitPerDay, result.currency)}
                </div>
              </div>
            </CardContent>
          </Card>

          {/* Detailed Results */}
          <Card>
            <CardHeader>
              <CardTitle className="flex items-center gap-2">
                <DollarSign className="h-5 w-5" />
                {t("detailedResults")}
              </CardTitle>
            </CardHeader>
            <CardContent>
              <div className="grid gap-4">
                {/* Revenue */}
                <div className="grid grid-cols-3 gap-4 text-center">
                  <div className="p-3 bg-muted rounded-lg">
                    <div className="text-xs text-muted-foreground">{t("daily")}</div>
                    <div className="font-medium">{formatMiningCurrency(result.revenuePerDay, result.currency)}</div>
                    <div className="text-xs text-muted-foreground">{formatBtc(result.btcPerDay)} BTC</div>
                  </div>
                  <div className="p-3 bg-muted rounded-lg">
                    <div className="text-xs text-muted-foreground">{t("monthly")}</div>
                    <div className="font-medium">{formatMiningCurrency(result.revenuePerMonth, result.currency)}</div>
                    <div className="text-xs text-muted-foreground">{formatBtc(result.btcPerMonth)} BTC</div>
                  </div>
                  <div className="p-3 bg-muted rounded-lg">
                    <div className="text-xs text-muted-foreground">{t("yearly")}</div>
                    <div className="font-medium">{formatMiningCurrency(result.revenuePerYear, result.currency)}</div>
                    <div className="text-xs text-muted-foreground">{formatBtc(result.btcPerYear)} BTC</div>
                  </div>
                </div>

                {/* Electricity Cost */}
                <div className="flex justify-between items-center py-2 border-b">
                  <span className="text-muted-foreground">{t("electricityCostDaily")}</span>
                  <span className="font-medium text-destructive">
                    -{formatMiningCurrency(result.electricityCostPerDay, result.currency)}
                  </span>
                </div>

                {/* ROI (if applicable) */}
                {result.roiDays && (
                  <>
                    <div className="flex justify-between items-center py-2 border-b">
                      <span className="text-muted-foreground">{t("roiDays")}</span>
                      <span className="font-medium">{result.roiDays} {t("days")}</span>
                    </div>
                    <div className="flex justify-between items-center py-2 border-b">
                      <span className="text-muted-foreground">{t("breakEvenDate")}</span>
                      <span className="font-medium">{result.breakEvenDate}</span>
                    </div>
                  </>
                )}

                {/* Network Info */}
                <div className="pt-4 space-y-2 text-sm text-muted-foreground">
                  <div className="flex justify-between">
                    <span>{t("btcPrice")}</span>
                    <span>{formatMiningCurrency(result.btcPrice, result.currency)}</span>
                  </div>
                  <div className="flex justify-between">
                    <span>{t("blockReward")}</span>
                    <span>{result.blockReward} BTC</span>
                  </div>
                </div>
              </div>
            </CardContent>
          </Card>

          {/* Data Freshness */}
          <div className="flex items-center gap-2 text-sm text-muted-foreground">
            <Clock className="h-4 w-4" />
            <span>
              {t("dataFrom")}: {result.source} | {t("lastUpdated")}: {new Date(lastUpdated).toLocaleDateString()}
              {dataAge > 0 && ` (${dataAge} ${t("hoursAgo")})`}
            </span>
          </div>
        </>
      )}

      {/* Reset Button */}
      <div className="flex justify-end">
        <Button variant="outline" onClick={reset}>
          <RotateCcw className="h-4 w-4 mr-2" />
          {commonT("reset")}
        </Button>
      </div>
    </div>
  );
}
