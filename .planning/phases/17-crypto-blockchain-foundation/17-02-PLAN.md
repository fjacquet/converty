import WAValidator from "wallet-address-validator";

export type WalletType = "BTC" | "ETH" | "LTC";

export interface AddressFormat {
  id: string;
  name: string;
  description: string;
}

export interface WalletValidationResult {
  address: string;
  walletType: WalletType;
  isValid: boolean;
  addressFormat: AddressFormat | null;
  networkType: "mainnet" | "testnet" | null;
  checksumValid: boolean | null;
  warningMessage: string | null;
}

/**
 * Address format definitions for each wallet type
 */
const ADDRESS_FORMATS: Record<WalletType, AddressFormat[]> = {
  BTC: [
    { id: "p2pkh", name: "P2PKH (Legacy)", description: "Pay-to-Public-Key-Hash, starts with 1" },
    { id: "p2sh", name: "P2SH", description: "Pay-to-Script-Hash, starts with 3" },
    { id: "p2wpkh", name: "P2WPKH (Native SegWit)", description: "Bech32, starts with bc1q" },
    { id: "p2wsh", name: "P2WSH", description: "Bech32, starts with bc1q (longer)" },
    { id: "p2tr", name: "P2TR (Taproot)", description: "Bech32m, starts with bc1p" },
  ],
  ETH: [
    { id: "erc20", name: "Ethereum (ERC-20)", description: "Standard Ethereum address, starts with 0x" },
  ],
  LTC: [
    { id: "legacy", name: "Legacy", description: "Starts with L or M" },
    { id: "segwit", name: "SegWit", description: "Starts with ltc1" },
  ],
};

/**
 * Detect Bitcoin address format from prefix
 */
function detectBitcoinFormat(address: string): AddressFormat | null {
  if (address.startsWith("1")) {
    return ADDRESS_FORMATS.BTC[0]; // P2PKH
  }
  if (address.startsWith("3")) {
    return ADDRESS_FORMATS.BTC[1]; // P2SH
  }
  if (address.startsWith("bc1q")) {
    // Distinguish P2WPKH (42 chars) from P2WSH (62 chars)
    return address.length <= 44 ? ADDRESS_FORMATS.BTC[2] : ADDRESS_FORMATS.BTC[3];
  }
  if (address.startsWith("bc1p")) {
    return ADDRESS_FORMATS.BTC[4]; // P2TR (Taproot)
  }
  return null;
}

/**
 * Detect Litecoin address format from prefix
 */
function detectLitecoinFormat(address: string): AddressFormat | null {
  if (address.startsWith("L") || address.startsWith("M")) {
    return ADDRESS_FORMATS.LTC[0]; // Legacy
  }
  if (address.startsWith("ltc1")) {
    return ADDRESS_FORMATS.LTC[1]; // SegWit
  }
  return null;
}

/**
 * Validate Ethereum ERC-55 checksum (mixed-case)
 */
function validateEthereumChecksum(address: string): boolean {
  // If all lowercase or all uppercase (except 0x prefix), checksum doesn't apply
  const addressWithoutPrefix = address.slice(2);
  const isAllLower = addressWithoutPrefix === addressWithoutPrefix.toLowerCase();
  const isAllUpper = addressWithoutPrefix === addressWithoutPrefix.toUpperCase();

  if (isAllLower || isAllUpper) {
    // Valid but no checksum applied
    return true;
  }

  // ERC-55 checksum validation would require keccak256 hash
  // For MVP, we accept mixed-case as potentially checksummed
  // Full implementation would need keccak256 library
  return true;
}

/**
 * Detect network type (mainnet vs testnet)
 */
function detectNetworkType(address: string, walletType: WalletType): "mainnet" | "testnet" | null {
  if (walletType === "BTC") {
    // Testnet addresses start with m, n, or 2 (legacy) or tb1 (segwit)
    if (address.startsWith("m") || address.startsWith("n") || address.startsWith("2") || address.startsWith("tb1")) {
      return "testnet";
    }
    return "mainnet";
  }
  if (walletType === "ETH") {
    // Ethereum doesn't have address-level mainnet/testnet distinction
    return null;
  }
  if (walletType === "LTC") {
    // Litecoin testnet uses different prefixes
    if (address.startsWith("tltc1") || address.startsWith("Q") || address.startsWith("m")) {
      return "testnet";
    }
    return "mainnet";
  }
  return null;
}

/**
 * Check if input might be a private key (security warning)
 */
function mightBePrivateKey(input: string): boolean {
  // Bitcoin WIF private keys start with 5, K, or L and are 51-52 characters
  if (/^[5KL][1-9A-HJ-NP-Za-km-z]{50,51}$/.test(input)) {
    return true;
  }
  // Ethereum private keys are 64 hex characters (sometimes with 0x prefix)
  if (/^(0x)?[a-fA-F0-9]{64}$/.test(input)) {
    return true;
  }
  return false;
}

/**
 * Validate wallet address
 */
export function validateWalletAddress(
  address: string,
  walletType: WalletType
): WalletValidationResult {
  const trimmedAddress = address.trim();

  // Check for potential private key input
  if (mightBePrivateKey(trimmedAddress)) {
    return {
      address: trimmedAddress,
      walletType,
      isValid: false,
      addressFormat: null,
      networkType: null,
      checksumValid: null,
      warningMessage: "privateKeyWarning",
    };
  }

  // Empty input
  if (!trimmedAddress) {
    return {
      address: "",
      walletType,
      isValid: false,
      addressFormat: null,
      networkType: null,
      checksumValid: null,
      warningMessage: null,
    };
  }

  // Validate using wallet-address-validator
  const isValid = WAValidator.validate(trimmedAddress, walletType);

  // Detect address format
  let addressFormat: AddressFormat | null = null;
  if (walletType === "BTC") {
    addressFormat = detectBitcoinFormat(trimmedAddress);
  } else if (walletType === "ETH") {
    addressFormat = trimmedAddress.startsWith("0x") ? ADDRESS_FORMATS.ETH[0] : null;
  } else if (walletType === "LTC") {
    addressFormat = detectLitecoinFormat(trimmedAddress);
  }

  // Detect network type
  const networkType = detectNetworkType(trimmedAddress, walletType);

  // Checksum validation (Ethereum only for now)
  let checksumValid: boolean | null = null;
  if (walletType === "ETH" && isValid) {
    checksumValid = validateEthereumChecksum(trimmedAddress);
  }

  return {
    address: trimmedAddress,
    walletType,
    isValid,
    addressFormat,
    networkType,
    checksumValid,
    warningMessage: networkType === "testnet" ? "testnetWarning" : null,
  };
}

/**
 * Get supported wallet types with descriptions
 */
export const WALLET_TYPES: { type: WalletType; name: string; symbol: string }[] = [
  { type: "BTC", name: "Bitcoin", symbol: "BTC" },
  { type: "ETH", name: "Ethereum", symbol: "ETH" },
  { type: "LTC", name: "Litecoin", symbol: "LTC" },
];

/**
 * Test addresses for verification
 */
export const TEST_ADDRESSES: Record<WalletType, { valid: string; invalid: string }> = {
  BTC: {
    valid: "bc1qar0srrr7xfkvy5l643lydnw9re59gtzzwf5mdq",
    invalid: "bc1invalid123456789",
  },
  ETH: {
    valid: "0x742d35Cc6634C0532925a3b844Bc454e4438f44e",
    invalid: "0xinvalidaddress",
  },
  LTC: {
    valid: "ltc1qw508d6qejxtdg4y5r3zarvary0c5xw7kgmn4n9",
    invalid: "ltc1invalid",
  },
};
```

// src/stores/wallet-validator-store.ts
"use client";

import { create } from "zustand";
import {
  validateWalletAddress,
  type WalletType,
  type WalletValidationResult,
} from "@/lib/converters/crypto/wallet-validator";
import { createUrlSyncMiddleware } from "@/lib/middleware/url-sync";
import { getUrlParams, parseStringParam } from "@/lib/utils/url-params";

export interface WalletValidatorState {
  // Input
  address: string;
  walletType: WalletType;

  // Result
  result: WalletValidationResult | null;
  error: string | null;

  // Actions
  setAddress: (value: string) => void;
  setWalletType: (type: WalletType) => void;
  validate: () => void;
  reset: () => void;
}

const initialState = {
  address: "",
  walletType: "BTC" as WalletType,
  result: null,
  error: null,
};

export const useWalletValidatorStore = create<WalletValidatorState>()(
  createUrlSyncMiddleware<WalletValidatorState>({
    enabled: true,
    debounceMs: 300,
    selectState: (state) => ({
      address: state.address,
      walletType: state.walletType,
    }),
  })((set, get) => {
    // Load initial values from URL params if present
    let loadedAddress = initialState.address;
    let loadedWalletType = initialState.walletType;

    if (typeof window !== "undefined") {
      const urlParams = getUrlParams();
      if (Object.keys(urlParams).length > 0) {
        loadedAddress = parseStringParam(urlParams.address, initialState.address);
        const typeParam = parseStringParam(urlParams.walletType, initialState.walletType);
        if (["BTC", "ETH", "LTC"].includes(typeParam)) {
          loadedWalletType = typeParam as WalletType;
        }
      }
    }

    return {
      address: loadedAddress,
      walletType: loadedWalletType,
      result: null,
      error: null,

      setAddress: (value: string) => {
        set({ address: value, error: null });
        // Auto-validate when address changes
        if (value.trim()) {
          setTimeout(() => {
            get().validate();
          }, 0);
        } else {
          set({ result: null });
        }
      },

      setWalletType: (walletType: WalletType) => {
        set({ walletType, error: null });
        // Re-validate if address exists
        const { address } = get();
        if (address.trim()) {
          setTimeout(() => {
            get().validate();
          }, 0);
        }
      },

      validate: () => {
        const { address, walletType } = get();
        if (!address.trim()) {
          set({ result: null, error: null });
          return;
        }

        try {
          const result = validateWalletAddress(address, walletType);
          set({ result, error: null });
        } catch (err) {
          set({
            result: null,
            error: err instanceof Error ? err.message : "Validation failed",
          });
        }
      },

      reset: () => set({ ...initialState, result: null }),
    };
  })
);
```

Key patterns:
- Sync only address and walletType to URL
- Auto-validate on input change
- Synchronous validation (no async needed)

"use client";

import { AlertTriangle, CheckCircle, Info, RotateCcw, Shield, Wallet, XCircle } from "lucide-react";
import { useTranslations } from "next-intl";
import { Alert, AlertDescription, AlertTitle } from "@/components/ui/alert";
import { Button } from "@/components/ui/button";
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";
import { Input } from "@/components/ui/input";
import { Label } from "@/components/ui/label";
import {
  Select,
  SelectContent,
  SelectItem,
  SelectTrigger,
  SelectValue,
} from "@/components/ui/select";
import { WALLET_TYPES, type WalletType } from "@/lib/converters/crypto/wallet-validator";
import { useWalletValidatorStore } from "@/stores/wallet-validator-store";

export function WalletValidatorCalculator() {
  const t = useTranslations("calculator.crypto.wallet");
  const commonT = useTranslations("common");
  const { address, walletType, result, error, setAddress, setWalletType, reset } =
    useWalletValidatorStore();

  return (
    <div className="space-y-6">
      {/* Security Notice */}
      <Alert>
        <Shield className="h-4 w-4" />
        <AlertTitle>{t("securityNotice")}</AlertTitle>
        <AlertDescription>{t("securityDescription")}</AlertDescription>
      </Alert>

      {/* Input Section */}
      <Card>
        <CardHeader>
          <CardTitle className="flex items-center gap-2">
            <Wallet className="h-5 w-5" />
            {t("input")}
          </CardTitle>
        </CardHeader>
        <CardContent className="space-y-4">
          <div className="space-y-2">
            <Label htmlFor="walletType">{t("walletType")}</Label>
            <Select value={walletType} onValueChange={(v) => setWalletType(v as WalletType)}>
              <SelectTrigger id="walletType">
                <SelectValue />
              </SelectTrigger>
              <SelectContent>
                {WALLET_TYPES.map((w) => (
                  <SelectItem key={w.type} value={w.type}>
                    {w.name} ({w.symbol})
                  </SelectItem>
                ))}
              </SelectContent>
            </Select>
          </div>

          <div className="space-y-2">
            <Label htmlFor="address">{t("addressLabel")}</Label>
            <Input
              id="address"
              value={address}
              onChange={(e) => setAddress(e.target.value)}
              placeholder={t("addressPlaceholder")}
              className="font-mono"
            />
            <p className="text-sm text-muted-foreground">{t("publicAddressOnly")}</p>
          </div>
        </CardContent>
      </Card>

      {/* Error Display */}
      {error && (
        <Card className="border-destructive">
          <CardContent className="pt-6">
            <p className="text-destructive">{error}</p>
          </CardContent>
        </Card>
      )}

      {/* Private Key Warning */}
      {result?.warningMessage === "privateKeyWarning" && (
        <Alert variant="destructive">
          <AlertTriangle className="h-4 w-4" />
          <AlertTitle>{t("warnings.privateKeyTitle")}</AlertTitle>
          <AlertDescription>{t("warnings.privateKeyDescription")}</AlertDescription>
        </Alert>
      )}

      {/* Testnet Warning */}
      {result?.warningMessage === "testnetWarning" && (
        <Alert>
          <Info className="h-4 w-4" />
          <AlertTitle>{t("warnings.testnetTitle")}</AlertTitle>
          <AlertDescription>{t("warnings.testnetDescription")}</AlertDescription>
        </Alert>
      )}

      {/* Results Section */}
      {result && result.address && (
        <Card>
          <CardHeader>
            <CardTitle className="flex items-center gap-2">
              {result.isValid ? (
                <CheckCircle className="h-5 w-5 text-green-500" />
              ) : (
                <XCircle className="h-5 w-5 text-destructive" />
              )}
              {t("result")}
            </CardTitle>
          </CardHeader>
          <CardContent className="space-y-4">
            {/* Validation Status */}
            <div className="flex items-center gap-2 p-4 rounded-md bg-muted">
              {result.isValid ? (
                <>
                  <CheckCircle className="h-6 w-6 text-green-500" />
                  <span className="font-medium text-green-700 dark:text-green-400">
                    {t("validAddress")}
                  </span>
                </>
              ) : (
                <>
                  <XCircle className="h-6 w-6 text-destructive" />
                  <span className="font-medium text-destructive">{t("invalidAddress")}</span>
                </>
              )}
            </div>

            {/* Address Details (if valid) */}
            {result.isValid && (
              <div className="grid gap-4">
                {/* Wallet Type */}
                <div className="flex justify-between items-center py-2 border-b">
                  <span className="text-muted-foreground">{t("walletTypeLabel")}</span>
                  <span className="font-medium">
                    {WALLET_TYPES.find((w) => w.type === result.walletType)?.name}
                  </span>
                </div>

                {/* Address Format */}
                {result.addressFormat && (
                  <div className="flex justify-between items-center py-2 border-b">
                    <span className="text-muted-foreground">{t("addressFormat")}</span>
                    <div className="text-right">
                      <span className="font-medium">{result.addressFormat.name}</span>
                      <p className="text-sm text-muted-foreground">
                        {result.addressFormat.description}
                      </p>
                    </div>
                  </div>
                )}

                {/* Network Type */}
                {result.networkType && (
                  <div className="flex justify-between items-center py-2 border-b">
                    <span className="text-muted-foreground">{t("networkType")}</span>
                    <span className="font-medium capitalize">{result.networkType}</span>
                  </div>
                )}

                {/* Checksum (Ethereum only) */}
                {result.checksumValid !== null && (
                  <div className="flex justify-between items-center py-2 border-b">
                    <span className="text-muted-foreground">{t("checksumValid")}</span>
                    <span className="font-medium">
                      {result.checksumValid ? t("yes") : t("no")}
                    </span>
                  </div>
                )}
              </div>
            )}
          </CardContent>
        </Card>
      )}

      {/* Reset Button */}
      <div className="flex justify-end">
        <Button variant="outline" onClick={reset}>
          <RotateCcw className="h-4 w-4 mr-2" />
          {commonT("reset")}
        </Button>
      </div>
    </div>
  );
}
