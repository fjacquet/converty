import type { SupportedCurrency } from "@/components/converter/currency-selector";
import benchmarks from "@/lib/data/real-estate-benchmarks.json";

export interface RentalYieldInput {
  purchasePrice: number;
  annualRent: number;
  annualExpenses: number; // Property tax, insurance, maintenance, management
  transactionCostsPercent: number; // Typically 3-6% in Switzerland
  currency: SupportedCurrency;
  // Optional mortgage for cash flow analysis
  includeMortgage: boolean;
  monthlyMortgagePayment: number;
}

export interface RentalYieldResult {
  // Core yields
  grossYield: number;
  netYield: number;

  // Investment metrics
  grm: number; // Gross Rent Multiplier
  capRate: number; // Capitalization rate (simplified)

  // Cash flow
  monthlyGrossIncome: number;
  monthlyExpenses: number;
  monthlyCashFlow: number;
  annualCashFlow: number;

  // Total investment
  totalInvestment: number;
  transactionCosts: number;
  netOperatingIncome: number;

  // Market comparison
  marketAverage: number;
  comparisonToMarket: "above" | "below" | "average";
  comparisonPercent: number;
  rating: "excellent" | "good" | "fair" | "poor";

  // Break-even
  yearsToBreakEven: number;

  currency: SupportedCurrency;
}

/**
 * Get Swiss benchmark values
 */
export function getSwissBenchmarks() {
  return {
    averageGrossYield: benchmarks.switzerland.averageGrossYield,
    averageNetYield: benchmarks.switzerland.averageNetYield,
    yields: benchmarks.switzerland.yields,
    transactionCosts: benchmarks.switzerland.transactionCosts,
    operatingCostPercent: benchmarks.switzerland.operatingCostPercent,
    ratingThresholds: benchmarks.ratingThresholds,
  };
}

/**
 * Get yield rating based on Swiss market thresholds
 */
function getYieldRating(netYield: number): "excellent" | "good" | "fair" | "poor" {
  const thresholds = benchmarks.ratingThresholds;
  if (netYield >= thresholds.excellentMinYield) return "excellent";
  if (netYield >= thresholds.goodMinYield) return "good";
  if (netYield >= thresholds.fairMinYield) return "fair";
  return "poor";
}

/**
 * Calculate rental yield and investment metrics
 */
export function calculateRentalYield(input: RentalYieldInput): RentalYieldResult | null {
  const {
    purchasePrice,
    annualRent,
    annualExpenses,
    transactionCostsPercent,
    currency,
    includeMortgage,
    monthlyMortgagePayment,
  } = input;

  // Validation
  if (purchasePrice <= 0) return null;
  if (annualRent < 0 || annualExpenses < 0) return null;

  // Calculate transaction costs
  const transactionCosts = purchasePrice * (transactionCostsPercent / 100);
  const totalInvestment = purchasePrice + transactionCosts;

  // Calculate yields
  const grossYield = annualRent > 0 ? (annualRent / purchasePrice) * 100 : 0;

  const netOperatingIncome = annualRent - annualExpenses;
  const netYield = annualRent > 0 ? (netOperatingIncome / totalInvestment) * 100 : 0;

  // Investment metrics
  const grm = annualRent > 0 ? purchasePrice / annualRent : 0;
  const capRate = purchasePrice > 0 ? (netOperatingIncome / purchasePrice) * 100 : 0;

  // Cash flow
  const monthlyGrossIncome = annualRent / 12;
  const monthlyExpenses = annualExpenses / 12;
  let monthlyCashFlow = monthlyGrossIncome - monthlyExpenses;

  if (includeMortgage && monthlyMortgagePayment > 0) {
    monthlyCashFlow -= monthlyMortgagePayment;
  }

  const annualCashFlow = monthlyCashFlow * 12;

  // Market comparison
  const marketAverage = benchmarks.switzerland.averageGrossYield;
  let comparisonToMarket: "above" | "below" | "average";
  const comparisonPercent = grossYield - marketAverage;

  if (Math.abs(comparisonPercent) < 0.2) {
    comparisonToMarket = "average";
  } else if (comparisonPercent > 0) {
    comparisonToMarket = "above";
  } else {
    comparisonToMarket = "below";
  }

  // Rating
  const rating = getYieldRating(netYield);

  // Break-even (years to recover investment from net income)
  const yearsToBreakEven = netOperatingIncome > 0
    ? totalInvestment / netOperatingIncome
    : Infinity;

  return {
    grossYield: Math.round(grossYield * 100) / 100,
    netYield: Math.round(netYield * 100) / 100,
    grm: Math.round(grm * 100) / 100,
    capRate: Math.round(capRate * 100) / 100,
    monthlyGrossIncome: Math.round(monthlyGrossIncome * 100) / 100,
    monthlyExpenses: Math.round(monthlyExpenses * 100) / 100,
    monthlyCashFlow: Math.round(monthlyCashFlow * 100) / 100,
    annualCashFlow: Math.round(annualCashFlow * 100) / 100,
    totalInvestment: Math.round(totalInvestment * 100) / 100,
    transactionCosts: Math.round(transactionCosts * 100) / 100,
    netOperatingIncome: Math.round(netOperatingIncome * 100) / 100,
    marketAverage,
    comparisonToMarket,
    comparisonPercent: Math.round(comparisonPercent * 100) / 100,
    rating,
    yearsToBreakEven: Math.round(yearsToBreakEven * 10) / 10,
    currency,
  };
}

/**
 * Get Swiss city yields for comparison display
 */
export function getSwissCityYields(): Record<string, number> {
  return benchmarks.switzerland.yields;
}

// src/stores/rental-yield-store.ts
"use client";

import { create } from "zustand";
import type { SupportedCurrency } from "@/components/converter/currency-selector";
import {
  calculateRentalYield,
  getSwissBenchmarks,
  type RentalYieldInput,
  type RentalYieldResult,
} from "@/lib/converters/realestate/rental-yield";
import { createUrlSyncMiddleware } from "@/lib/middleware/url-sync";
import { getUrlParams, parseStringParam, parseNumberParam, parseBooleanParam } from "@/lib/utils/url-params";

export interface RentalYieldState {
  // Input values
  purchasePrice: number;
  annualRent: number;
  annualExpenses: number;
  transactionCostsPercent: number;
  currency: SupportedCurrency;
  includeMortgage: boolean;
  monthlyMortgagePayment: number;

  // Result
  result: RentalYieldResult | null;
  error: string | null;

  // Actions
  setPurchasePrice: (value: number) => void;
  setAnnualRent: (value: number) => void;
  setMonthlyRent: (value: number) => void;
  setAnnualExpenses: (value: number) => void;
  setTransactionCostsPercent: (value: number) => void;
  setCurrency: (value: SupportedCurrency) => void;
  setIncludeMortgage: (value: boolean) => void;
  setMonthlyMortgagePayment: (value: number) => void;
  calculate: () => void;
  reset: () => void;
}

const benchmarks = getSwissBenchmarks();

const initialState = {
  purchasePrice: 800000,
  annualRent: 24000, // 2000/month
  annualExpenses: 4000, // ~0.5% of property value
  transactionCostsPercent: benchmarks.transactionCosts.typicalPercent,
  currency: "CHF" as SupportedCurrency,
  includeMortgage: false,
  monthlyMortgagePayment: 0,
  result: null,
  error: null,
};

export const useRentalYieldStore = create<RentalYieldState>()(
  createUrlSyncMiddleware<RentalYieldState>({
    enabled: true,
    debounceMs: 300,
    selectState: (state) => ({
      purchasePrice: state.purchasePrice,
      annualRent: state.annualRent,
      annualExpenses: state.annualExpenses,
      transactionCostsPercent: state.transactionCostsPercent,
      currency: state.currency,
      includeMortgage: state.includeMortgage,
      monthlyMortgagePayment: state.monthlyMortgagePayment,
    }),
  })((set, get) => {
    // Load from URL params
    let loaded = { ...initialState };

    if (typeof window !== "undefined") {
      const urlParams = getUrlParams();
      if (Object.keys(urlParams).length > 0) {
        loaded.purchasePrice = parseNumberParam(urlParams.purchasePrice, initialState.purchasePrice);
        loaded.annualRent = parseNumberParam(urlParams.annualRent, initialState.annualRent);
        loaded.annualExpenses = parseNumberParam(urlParams.annualExpenses, initialState.annualExpenses);
        loaded.transactionCostsPercent = parseNumberParam(urlParams.transactionCostsPercent, initialState.transactionCostsPercent);
        loaded.monthlyMortgagePayment = parseNumberParam(urlParams.monthlyMortgagePayment, initialState.monthlyMortgagePayment);

        const currencyParam = parseStringParam(urlParams.currency, initialState.currency);
        if (["CHF", "EUR"].includes(currencyParam)) {
          loaded.currency = currencyParam as SupportedCurrency;
        }

        loaded.includeMortgage = parseBooleanParam(urlParams.includeMortgage, initialState.includeMortgage);
      }
    }

    return {
      ...loaded,

      setPurchasePrice: (value: number) => {
        set({ purchasePrice: value });
        setTimeout(() => get().calculate(), 0);
      },

      setAnnualRent: (value: number) => {
        set({ annualRent: value });
        setTimeout(() => get().calculate(), 0);
      },

      setMonthlyRent: (value: number) => {
        set({ annualRent: value * 12 });
        setTimeout(() => get().calculate(), 0);
      },

      setAnnualExpenses: (value: number) => {
        set({ annualExpenses: value });
        setTimeout(() => get().calculate(), 0);
      },

      setTransactionCostsPercent: (value: number) => {
        set({ transactionCostsPercent: value });
        setTimeout(() => get().calculate(), 0);
      },

      setCurrency: (value: SupportedCurrency) => {
        set({ currency: value });
        setTimeout(() => get().calculate(), 0);
      },

      setIncludeMortgage: (value: boolean) => {
        set({ includeMortgage: value });
        setTimeout(() => get().calculate(), 0);
      },

      setMonthlyMortgagePayment: (value: number) => {
        set({ monthlyMortgagePayment: value });
        setTimeout(() => get().calculate(), 0);
      },

      calculate: () => {
        const state = get();
        const input: RentalYieldInput = {
          purchasePrice: state.purchasePrice,
          annualRent: state.annualRent,
          annualExpenses: state.annualExpenses,
          transactionCostsPercent: state.transactionCostsPercent,
          currency: state.currency,
          includeMortgage: state.includeMortgage,
          monthlyMortgagePayment: state.monthlyMortgagePayment,
        };

        try {
          const result = calculateRentalYield(input);
          set({ result, error: null });
        } catch (err) {
          set({
            result: null,
            error: err instanceof Error ? err.message : "Calculation failed",
          });
        }
      },

      reset: () => set({ ...initialState }),
    };
  })
);

                    </span>
                  )}
                  {result.comparisonToMarket === "average" && (
                    <span className="flex items-center text-yellow-600 text-sm">
                      <Minus className="h-4 w-4 mr-1" />
                      {t("atAverage")}
                    </span>
                  )}
                </div>
              </div>

              {/* Rating Badge */}
              {(() => {
                const style = getRatingStyle(result.rating);
                const Icon = style.icon;
                return (
                  <div className={`inline-flex items-center gap-2 px-3 py-1 rounded-full ${style.bg}`}>
                    <Icon className={`h-4 w-4 ${style.color}`} />
                    <span className={`font-medium ${style.color}`}>
                      {t(`ratings.${result.rating}`)}
                    </span>
                  </div>
                );
              })()}
            </CardContent>
          </Card>

          {/* Investment Metrics */}
          <Card>
            <CardHeader>
              <CardTitle>{t("investmentMetrics")}</CardTitle>
            </CardHeader>
            <CardContent>
              <div className="grid grid-cols-2 md:grid-cols-4 gap-4">
                <div className="text-center p-4 bg-muted rounded-lg">
                  <p className="text-sm text-muted-foreground">{t("grm")}</p>
                  <p className="text-2xl font-bold">{result.grm}</p>
                  <p className="text-xs text-muted-foreground">{t("years")}</p>
                </div>
                <div className="text-center p-4 bg-muted rounded-lg">
                  <p className="text-sm text-muted-foreground">{t("capRate")}</p>
                  <p className="text-2xl font-bold">{result.capRate}%</p>
                </div>
                <div className="text-center p-4 bg-muted rounded-lg">
                  <p className="text-sm text-muted-foreground">{t("breakEven")}</p>
                  <p className="text-2xl font-bold">
                    {result.yearsToBreakEven === Infinity ? "-" : result.yearsToBreakEven}
                  </p>
                  <p className="text-xs text-muted-foreground">{t("years")}</p>
                </div>
                <div className="text-center p-4 bg-muted rounded-lg">
                  <p className="text-sm text-muted-foreground">{t("noi")}</p>
                  <p className="text-2xl font-bold">{formatCurrency(result.netOperatingIncome)}</p>
                  <p className="text-xs text-muted-foreground">{t("perYear")}</p>
                </div>
              </div>
            </CardContent>
          </Card>

          {/* Cash Flow */}
          <Card>
            <CardHeader>
              <CardTitle>{t("cashFlow")}</CardTitle>
            </CardHeader>
            <CardContent>
              <div className="space-y-4">
                <div className="flex justify-between items-center py-2 border-b">
                  <span>{t("monthlyGrossIncome")}</span>
                  <span className="font-medium text-green-600">
                    +{formatCurrency(result.monthlyGrossIncome)}
                  </span>
                </div>
                <div className="flex justify-between items-center py-2 border-b">
                  <span>{t("monthlyExpenses")}</span>
                  <span className="font-medium text-red-600">
                    -{formatCurrency(result.monthlyExpenses)}
                  </span>
                </div>
                {includeMortgage && monthlyMortgagePayment > 0 && (
                  <div className="flex justify-between items-center py-2 border-b">
                    <span>{t("mortgagePayment")}</span>
                    <span className="font-medium text-red-600">
                      -{formatCurrency(monthlyMortgagePayment)}
                    </span>
                  </div>
                )}
                <div className="flex justify-between items-center py-2 bg-muted px-4 rounded-lg">
                  <span className="font-medium">{t("monthlyCashFlow")}</span>
                  <span
                    className={`text-xl font-bold ${
                      result.monthlyCashFlow >= 0 ? "text-green-600" : "text-red-600"
                    }`}
                  >
                    {result.monthlyCashFlow >= 0 ? "+" : ""}
                    {formatCurrency(result.monthlyCashFlow)}
                  </span>
                </div>
                <div className="flex justify-between items-center py-2">
                  <span>{t("annualCashFlow")}</span>
                  <span
                    className={`font-medium ${
                      result.annualCashFlow >= 0 ? "text-green-600" : "text-red-600"
                    }`}
                  >
                    {result.annualCashFlow >= 0 ? "+" : ""}
                    {formatCurrency(result.annualCashFlow)}
                  </span>
                </div>
              </div>
            </CardContent>
          </Card>

          {/* Investment Summary */}
          <Card>
            <CardHeader>
              <CardTitle>{t("investmentSummary")}</CardTitle>
            </CardHeader>
            <CardContent>
              <div className="grid grid-cols-1 md:grid-cols-3 gap-4 text-sm">
                <div className="flex justify-between">
                  <span className="text-muted-foreground">{t("purchasePrice")}:</span>
                  <span className="font-medium">{formatCurrency(purchasePrice)}</span>
                </div>
                <div className="flex justify-between">
                  <span className="text-muted-foreground">{t("transactionCosts")}:</span>
                  <span className="font-medium">{formatCurrency(result.transactionCosts)}</span>
                </div>
                <div className="flex justify-between">
                  <span className="text-muted-foreground">{t("totalInvestment")}:</span>
                  <span className="font-medium">{formatCurrency(result.totalInvestment)}</span>
                </div>
              </div>
            </CardContent>
          </Card>

          {/* Swiss City Comparison */}
          <Card>
            <CardHeader>
              <CardTitle>{t("swissCityComparison")}</CardTitle>
            </CardHeader>
            <CardContent>
              <div className="grid grid-cols-2 md:grid-cols-4 gap-4">
                {Object.entries(cityYields).map(([city, yieldValue]) => (
                  <div key={city} className="text-center p-3 bg-muted rounded-lg">
                    <p className="text-sm font-medium capitalize">{t(`cities.${city}`)}</p>
                    <p className="text-lg font-bold">{yieldValue}%</p>
                  </div>
                ))}
              </div>
            </CardContent>
          </Card>
        </>
      )}

      {/* Negative Cash Flow Warning */}
      {result && result.monthlyCashFlow < 0 && (
        <Alert variant="destructive">
          <AlertTriangle className="h-4 w-4" />
          <AlertTitle>{t("negativeCashFlowTitle")}</AlertTitle>
          <AlertDescription>{t("negativeCashFlowDescription")}</AlertDescription>
        </Alert>
      )}

      {/* Reset Button */}
      <div className="flex justify-end">
        <Button variant="outline" onClick={reset}>
          <RotateCcw className="h-4 w-4 mr-2" />
          {commonT("reset")}
        </Button>
      </div>
    </div>
  );
}
