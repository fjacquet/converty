3. Use store in component:

```typescript
  const { values, setValue, result } = useAgeStore();

  return (
    <div className="space-y-6">
      <InputField
        id="birthDate"
        label="Birth Date"
        type="date"
        value={values.birthDate}
        onChange={(value) => setValue("birthDate", value)}
      />

      {result && (

          results={[

            { label: "Months", value: result.months },
          ]}
        />
      )}
    </div>
  );
}
```
