  1. Create `src/lib/converters/engineering/types.ts` with interfaces:
     ```typescript
     /**
      * Engineering material properties
      * Sources: AISC Steel Construction Manual 15th Ed, ASM Handbook, Wood Design Manual
      */
     export interface Material {
       /** Unique identifier (kebab-case) */
       id: string;
       /** Display name */
       name: string;
       /** Material category: steel, aluminum, wood, concrete, other */
       category: "steel" | "aluminum" | "wood" | "concrete" | "other";
       /** Young's modulus in GPa */
       youngsModulus: number;
       /** Yield strength in MPa (null for materials like concrete that use compressive strength) */
       yieldStrength: number | null;
       /** Ultimate tensile strength in MPa */
       ultimateStrength: number | null;
       /** Density in kg/m3 */
       density: number;
       /** Poisson's ratio (dimensionless) */
       poissonRatio: number;
       /** Compressive strength in MPa (for concrete and wood) */
       compressiveStrength?: number;
       /** Standard reference (e.g., "ASTM A36", "EN 1992-1-1") */
       standard?: string;
     }

     /**
      * Standard beam cross-section properties
      * Source: AISC Steel Construction Manual 15th Edition
      */
     export interface BeamSection {
       /** Unique identifier (e.g., "W310x38") */
       id: string;
       /** Display name */
       name: string;
       /** Section type */
       type: "w-shape" | "c-channel" | "angle" | "hss-rect" | "hss-round" | "solid-rect" | "solid-circle";
       /** Cross-sectional area in mm2 */
       area: number;
       /** Moment of inertia about x-axis (strong axis) in mm4 */
       Ix: number;
       /** Moment of inertia about y-axis (weak axis) in mm4 */
       Iy: number;
       /** Section modulus about x-axis in mm3 */
       Zx: number;
       /** Section modulus about y-axis in mm3 */
       Zy: number;
       /** Depth in mm */
       depth: number;
       /** Flange width in mm */
       flangeWidth: number;
       /** Web thickness in mm */
       webThickness: number;
       /** Flange thickness in mm */
       flangeThickness: number;
       /** Mass per meter in kg/m */
       massPerMeter: number;
     }
     ```
