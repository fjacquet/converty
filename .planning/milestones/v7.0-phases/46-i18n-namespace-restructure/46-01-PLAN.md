    After the migration, verify structural parity: every key path in en.json must exist in fr.json, de.json, and it.json.

    Run the parity check script:
    ```bash
    node -e "
    const fs = require('fs');
    function flatKeys(obj, prefix = '') {
      const keys = [];
      for (const [k, v] of Object.entries(obj)) {
        const path = prefix ? prefix + '.' + k : k;
        if (typeof v === 'object' && v !== null) {
          keys.push(...flatKeys(v, path));
        } else {
          keys.push(path);
        }
      }
      return keys;
    }

    const locales = ['en', 'fr', 'de', 'it'];
    const data = {};
    for (const l of locales) {
      data[l] = JSON.parse(fs.readFileSync('src/messages/' + l + '.json'));
    }

    const enKeys = new Set(flatKeys(data['en']));
    let allOk = true;

    for (const l of ['fr', 'de', 'it']) {
      const lKeys = new Set(flatKeys(data[l]));
      const missing = [...enKeys].filter(k => !lKeys.has(k));
      const extra = [...lKeys].filter(k => !enKeys.has(k));
      if (missing.length || extra.length) {
        allOk = false;
        console.log(l + ' MISSING: ' + missing.slice(0,10).join(', '));
        if (missing.length > 10) console.log('  ... and ' + (missing.length-10) + ' more');
        if (extra.length) console.log(l + ' EXTRA: ' + extra.slice(0,5).join(', '));
      } else {
        console.log(l + ' OK: ' + lKeys.size + ' keys match en.json');
      }
    }
    if (allOk) console.log('ALL LOCALES IN PARITY');
    "
    ```

    If any locale has missing keys, the migration script must have missed that locale — re-run it or manually patch the affected file. Key parity is required before Plan 02 can proceed.
