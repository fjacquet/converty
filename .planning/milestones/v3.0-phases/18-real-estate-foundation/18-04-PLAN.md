import type { SupportedCurrency } from "@/components/converter/currency-selector";
import propertyPrices from "@/lib/data/swiss-property-prices.json";

export type PropertyType = "apartment" | "house" | "commercial";
export type PropertyCondition = "poor" | "fair" | "good" | "very_good" | "excellent";
export type SwissRegion = "zurich" | "geneva" | "bern" | "basel" | "lausanne" | "lucerne" | "zug" | "national_average";

export interface PropertyFeature {
  id: string;
  bonus: number; // CHF bonus for this feature
}

export interface PropertyValuationInput {
  propertyType: PropertyType;
  region: SwissRegion;
  size: number; // m²
  rooms: number;
  constructionYear: number;
  condition: PropertyCondition;
  features: string[]; // feature IDs
  currency: SupportedCurrency;
}

export interface PropertyValuationResult {
  // Value range
  estimatedValue: number;
  minValue: number;
  maxValue: number;
  pricePerM2: number;

  // Breakdown
  baseValue: number;
  ageAdjustment: number;
  conditionAdjustment: number;
  featureBonus: number;
  totalAdjustments: number;

  // Context
  regionPricePerM2: number;
  regionName: string;
  confidence: "low" | "medium" | "high";
  currency: SupportedCurrency;

  // Comparison
  vsRegionalAverage: number; // percentage above/below average
}

/**
 * Property features with CHF bonuses
 */
export const PROPERTY_FEATURES: PropertyFeature[] = [
  { id: "balcony", bonus: 15000 },
  { id: "terrace", bonus: 25000 },
  { id: "garden", bonus: 40000 },
  { id: "parking", bonus: 30000 },
  { id: "garage", bonus: 50000 },
  { id: "elevator", bonus: 20000 },
  { id: "renovated", bonus: 35000 },
  { id: "view", bonus: 30000 },
  { id: "quiet", bonus: 20000 },
  { id: "modern_kitchen", bonus: 25000 },
  { id: "fireplace", bonus: 15000 },
];

/**
 * Condition multipliers
 */
const CONDITION_MULTIPLIERS: Record<PropertyCondition, number> = {
  poor: 0.75,
  fair: 0.90,
  good: 1.00,
  very_good: 1.10,
  excellent: 1.20,
};

/**
 * Get Swiss regions with their base prices
 */
export function getSwissRegions(): {
  id: SwissRegion;
  code: string;
  name: string;
  apartment: number;
  house: number;
  commercial: number;
}[] {
  const regions = propertyPrices.regions as Record<
    string,
    { code: string; apartment: number; house: number; commercial: number }
  >;

  return Object.entries(regions).map(([id, data]) => ({
    id: id as SwissRegion,
    code: data.code,
    name: id.charAt(0).toUpperCase() + id.slice(1).replace("_", " "),
    apartment: data.apartment,
    house: data.house,
    commercial: data.commercial,
  }));
}

/**
 * Calculate age adjustment factor
 * New buildings (< 5 years): premium
 * Middle-aged (5-30 years): neutral
 * Old (30-60 years): slight discount
 * Historic (> 60 years): varies
 */
function calculateAgeAdjustment(constructionYear: number): number {
  const currentYear = new Date().getFullYear();
  const age = currentYear - constructionYear;

  if (age < 0) return 1.0; // Future construction, neutral
  if (age <= 5) return 1.10; // New build premium
  if (age <= 15) return 1.05; // Near-new
  if (age <= 30) return 1.00; // Prime age
  if (age <= 50) return 0.95; // Aging
  if (age <= 80) return 0.90; // Old
  return 0.85; // Very old (may have historical value, but default conservative)
}

/**
 * Calculate confidence level based on input completeness
 */
function calculateConfidence(input: PropertyValuationInput): "low" | "medium" | "high" {
  let score = 0;

  // Size validation
  if (input.size >= 20 && input.size <= 500) score += 1;

  // Rooms validation
  if (input.rooms >= 1 && input.rooms <= 10) score += 1;

  // Year validation
  const currentYear = new Date().getFullYear();
  if (input.constructionYear >= 1800 && input.constructionYear <= currentYear + 5) score += 1;

  // Features provided
  if (input.features.length > 0) score += 1;

  // Region is not national average
  if (input.region !== "national_average") score += 1;

  if (score >= 4) return "high";
  if (score >= 2) return "medium";
  return "low";
}

/**
 * Calculate property valuation using hedonic method
 */
export function calculatePropertyValuation(
  input: PropertyValuationInput
): PropertyValuationResult | null {
  const {
    propertyType,
    region,
    size,
    rooms,
    constructionYear,
    condition,
    features,
    currency,
  } = input;

  // Validation
  if (size <= 0) return null;

  // Get regional base price per m²
  const regions = propertyPrices.regions as Record<
    string,
    { code: string; apartment: number; house: number; commercial: number }
  >;
  const regionData = regions[region];
  if (!regionData) return null;

  const regionPricePerM2 = regionData[propertyType];
  if (!regionPricePerM2) return null;

  // Base value calculation
  const baseValue = size * regionPricePerM2;

  // Age adjustment
  const ageMultiplier = calculateAgeAdjustment(constructionYear);
  const ageAdjustment = baseValue * (ageMultiplier - 1);

  // Condition adjustment
  const conditionMultiplier = CONDITION_MULTIPLIERS[condition];
  const conditionAdjustment = baseValue * (conditionMultiplier - 1);

  // Feature bonuses
  let featureBonus = 0;
  for (const featureId of features) {
    const feature = PROPERTY_FEATURES.find((f) => f.id === featureId);
    if (feature) {
      featureBonus += feature.bonus;
    }
  }

  // Cap feature bonus at 20% of base value to prevent unrealistic values
  const maxFeatureBonus = baseValue * 0.20;
  featureBonus = Math.min(featureBonus, maxFeatureBonus);

  // Total adjustments
  const totalAdjustments = ageAdjustment + conditionAdjustment + featureBonus;

  // Estimated value
  const estimatedValue = baseValue + totalAdjustments;

  // Value range (±15%)
  const minValue = estimatedValue * 0.85;
  const maxValue = estimatedValue * 1.15;

  // Price per m²
  const pricePerM2 = estimatedValue / size;

  // Comparison to regional average
  const nationalAverage = regions.national_average[propertyType];
  const vsRegionalAverage = ((pricePerM2 - nationalAverage) / nationalAverage) * 100;

  // Confidence level
  const confidence = calculateConfidence(input);

  // Apply currency conversion if EUR (simplified: 1 CHF ≈ 1.05 EUR historically)
  const currencyMultiplier = currency === "EUR" ? 1.05 : 1;

  return {
    estimatedValue: Math.round(estimatedValue * currencyMultiplier / 1000) * 1000,
    minValue: Math.round(minValue * currencyMultiplier / 1000) * 1000,
    maxValue: Math.round(maxValue * currencyMultiplier / 1000) * 1000,
    pricePerM2: Math.round(pricePerM2 * currencyMultiplier),
    baseValue: Math.round(baseValue * currencyMultiplier),
    ageAdjustment: Math.round(ageAdjustment * currencyMultiplier),
    conditionAdjustment: Math.round(conditionAdjustment * currencyMultiplier),
    featureBonus: Math.round(featureBonus * currencyMultiplier),
    totalAdjustments: Math.round(totalAdjustments * currencyMultiplier),
    regionPricePerM2: Math.round(regionPricePerM2 * currencyMultiplier),
    regionName: region.replace("_", " ").replace(/\b\w/g, (l) => l.toUpperCase()),
    confidence,
    currency,
    vsRegionalAverage: Math.round(vsRegionalAverage * 10) / 10,
  };
}

/**
 * Get available property types
 */
export function getPropertyTypes(): PropertyType[] {
  return ["apartment", "house", "commercial"];
}

/**
 * Get available conditions
 */
export function getConditions(): PropertyCondition[] {
  return ["poor", "fair", "good", "very_good", "excellent"];
}

// src/stores/property-valuation-store.ts
"use client";

import { create } from "zustand";
import type { SupportedCurrency } from "@/components/converter/currency-selector";
import {
  calculatePropertyValuation,
  type PropertyType,
  type PropertyCondition,
  type SwissRegion,
  type PropertyValuationInput,
  type PropertyValuationResult,
} from "@/lib/converters/realestate/property-valuation";
import { createUrlSyncMiddleware } from "@/lib/middleware/url-sync";
import { getUrlParams, parseStringParam, parseNumberParam } from "@/lib/utils/url-params";

export interface PropertyValuationState {
  // Input values
  propertyType: PropertyType;
  region: SwissRegion;
  size: number;
  rooms: number;
  constructionYear: number;
  condition: PropertyCondition;
  features: string[];
  currency: SupportedCurrency;

  // Result
  result: PropertyValuationResult | null;
  error: string | null;

  // Actions
  setPropertyType: (value: PropertyType) => void;
  setRegion: (value: SwissRegion) => void;
  setSize: (value: number) => void;
  setRooms: (value: number) => void;
  setConstructionYear: (value: number) => void;
  setCondition: (value: PropertyCondition) => void;
  toggleFeature: (featureId: string) => void;
  setFeatures: (features: string[]) => void;
  setCurrency: (value: SupportedCurrency) => void;
  calculate: () => void;
  reset: () => void;
}

const currentYear = new Date().getFullYear();

const initialState = {
  propertyType: "apartment" as PropertyType,
  region: "zurich" as SwissRegion,
  size: 100,
  rooms: 4,
  constructionYear: 2015,
  condition: "good" as PropertyCondition,
  features: [] as string[],
  currency: "CHF" as SupportedCurrency,
  result: null,
  error: null,
};

export const usePropertyValuationStore = create<PropertyValuationState>()(
  createUrlSyncMiddleware<PropertyValuationState>({
    enabled: true,
    debounceMs: 300,
    selectState: (state) => ({
      propertyType: state.propertyType,
      region: state.region,
      size: state.size,
      rooms: state.rooms,
      constructionYear: state.constructionYear,
      condition: state.condition,
      features: state.features,
      currency: state.currency,
    }),
  })((set, get) => {
    // Load from URL params
    let loaded = { ...initialState };

    if (typeof window !== "undefined") {
      const urlParams = getUrlParams();
      if (Object.keys(urlParams).length > 0) {
        loaded.size = parseNumberParam(urlParams.size, initialState.size);
        loaded.rooms = parseNumberParam(urlParams.rooms, initialState.rooms);
        loaded.constructionYear = parseNumberParam(urlParams.constructionYear, initialState.constructionYear);

        const propertyTypeParam = parseStringParam(urlParams.propertyType, initialState.propertyType);
        if (["apartment", "house", "commercial"].includes(propertyTypeParam)) {
          loaded.propertyType = propertyTypeParam as PropertyType;
        }

        const regionParam = parseStringParam(urlParams.region, initialState.region);
        if (["zurich", "geneva", "bern", "basel", "lausanne", "lucerne", "zug", "national_average"].includes(regionParam)) {
          loaded.region = regionParam as SwissRegion;
        }

        const conditionParam = parseStringParam(urlParams.condition, initialState.condition);
        if (["poor", "fair", "good", "very_good", "excellent"].includes(conditionParam)) {
          loaded.condition = conditionParam as PropertyCondition;
        }

        const currencyParam = parseStringParam(urlParams.currency, initialState.currency);
        if (["CHF", "EUR"].includes(currencyParam)) {
          loaded.currency = currencyParam as SupportedCurrency;
        }

        // Features from comma-separated string
        if (urlParams.features) {
          loaded.features = String(urlParams.features).split(",").filter(Boolean);
        }
      }
    }

    return {
      ...loaded,

      setPropertyType: (value: PropertyType) => {
        set({ propertyType: value });
        setTimeout(() => get().calculate(), 0);
      },

      setRegion: (value: SwissRegion) => {
        set({ region: value });
        setTimeout(() => get().calculate(), 0);
      },

      setSize: (value: number) => {
        set({ size: value });
        setTimeout(() => get().calculate(), 0);
      },

      setRooms: (value: number) => {
        set({ rooms: value });
        setTimeout(() => get().calculate(), 0);
      },

      setConstructionYear: (value: number) => {
        set({ constructionYear: value });
        setTimeout(() => get().calculate(), 0);
      },

      setCondition: (value: PropertyCondition) => {
        set({ condition: value });
        setTimeout(() => get().calculate(), 0);
      },

      toggleFeature: (featureId: string) => {
        const { features } = get();
        const newFeatures = features.includes(featureId)
          ? features.filter((f) => f !== featureId)
          : [...features, featureId];
        set({ features: newFeatures });
        setTimeout(() => get().calculate(), 0);
      },

      setFeatures: (features: string[]) => {
        set({ features });
        setTimeout(() => get().calculate(), 0);
      },

      setCurrency: (value: SupportedCurrency) => {
        set({ currency: value });
        setTimeout(() => get().calculate(), 0);
      },

      calculate: () => {
        const state = get();
        const input: PropertyValuationInput = {
          propertyType: state.propertyType,
          region: state.region,
          size: state.size,
          rooms: state.rooms,
          constructionYear: state.constructionYear,
          condition: state.condition,
          features: state.features,
          currency: state.currency,
        };

        try {
          const result = calculatePropertyValuation(input);
          set({ result, error: null });
        } catch (err) {
          set({
            result: null,
            error: err instanceof Error ? err.message : "Calculation failed",
          });
        }
      },

      reset: () => set({ ...initialState }),
    };
  })
);

              />
            </div>
          </div>

          <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
            <div className="space-y-2">
              <Label htmlFor="constructionYear">{t("constructionYear")}</Label>
              <Input
                id="constructionYear"
                type="number"
                value={constructionYear}
                onChange={(e) => setConstructionYear(Number(e.target.value) || 2000)}
                min={1800}
                max={currentYear + 5}
              />
            </div>
            <div className="space-y-2">
              <Label htmlFor="condition">{t("condition")}</Label>
              <Select
                value={condition}
                onValueChange={(v) => setCondition(v as PropertyCondition)}
              >
                <SelectTrigger id="condition">
                  <SelectValue />
                </SelectTrigger>
                <SelectContent>
                  {conditions.map((cond) => (
                    <SelectItem key={cond} value={cond}>
                      {t(`conditions.${cond}`)}
                    </SelectItem>
                  ))}
                </SelectContent>
              </Select>
            </div>
          </div>
        </CardContent>
      </Card>

      {/* Features */}
      <Card>
        <CardHeader>
          <CardTitle className="flex items-center gap-2">
            <CheckCircle2 className="h-5 w-5" />
            {t("features")}
          </CardTitle>
        </CardHeader>
        <CardContent>
          <p className="text-sm text-muted-foreground mb-4">{t("featuresHint")}</p>
          <div className="grid grid-cols-2 md:grid-cols-3 lg:grid-cols-4 gap-4">
            {PROPERTY_FEATURES.map((feature) => (
              <div key={feature.id} className="flex items-center space-x-2">
                <Checkbox
                  id={feature.id}
                  checked={features.includes(feature.id)}
                  onCheckedChange={() => toggleFeature(feature.id)}
                />
                <Label htmlFor={feature.id} className="cursor-pointer text-sm">
                  {t(`featureNames.${feature.id}`)}
                </Label>
              </div>
            ))}
          </div>
        </CardContent>
      </Card>

      {/* Results */}
      {result && (
        <>
          {/* Main Valuation Result */}
          <Card>
            <CardHeader>
              <CardTitle className="flex items-center justify-between">
                <span>{t("estimatedValue")}</span>
                <TooltipProvider>
                  <Tooltip>
                    <TooltipTrigger asChild>
                      <div
                        className={`inline-flex items-center gap-1 px-2 py-1 rounded text-sm ${
                          getConfidenceStyle(result.confidence).bg
                        } ${getConfidenceStyle(result.confidence).color}`}
                      >
                        <HelpCircle className="h-4 w-4" />
                        {t(`confidence.${result.confidence}`)}
                      </div>
                    </TooltipTrigger>
                    <TooltipContent>
                      <p>{t("confidenceExplanation")}</p>
                    </TooltipContent>
                  </Tooltip>
                </TooltipProvider>
              </CardTitle>
            </CardHeader>
            <CardContent>
              {/* Value Range Display */}
              <div className="text-center p-6 bg-primary/10 rounded-lg mb-6">
                <p className="text-sm text-muted-foreground mb-2">{t("valueRange")}</p>
                <div className="flex items-center justify-center gap-4">
                  <div className="text-lg text-muted-foreground">
                    {formatCurrency(result.minValue)}
                  </div>
                  <div className="text-4xl font-bold text-primary">
                    {formatCurrency(result.estimatedValue)}
                  </div>
                  <div className="text-lg text-muted-foreground">
                    {formatCurrency(result.maxValue)}
                  </div>
                </div>
                <p className="text-xs text-muted-foreground mt-2">{t("rangeNote")}</p>
              </div>

              {/* Key Metrics */}
              <div className="grid grid-cols-2 md:grid-cols-3 gap-4">
                <div className="text-center p-4 bg-muted rounded-lg">
                  <p className="text-sm text-muted-foreground">{t("pricePerM2")}</p>
                  <p className="text-2xl font-bold">{formatCurrency(result.pricePerM2)}</p>
                  <p className="text-xs text-muted-foreground">/m²</p>
                </div>
                <div className="text-center p-4 bg-muted rounded-lg">
                  <p className="text-sm text-muted-foreground">{t("regionAverage")}</p>
                  <p className="text-2xl font-bold">{formatCurrency(result.regionPricePerM2)}</p>
                  <p className="text-xs text-muted-foreground">/m²</p>
                </div>
                <div className="text-center p-4 bg-muted rounded-lg">
                  <p className="text-sm text-muted-foreground">{t("vsNationalAverage")}</p>
                  <p
                    className={`text-2xl font-bold ${
                      result.vsRegionalAverage >= 0 ? "text-green-600" : "text-red-600"
                    }`}
                  >
                    {result.vsRegionalAverage >= 0 ? "+" : ""}
                    {result.vsRegionalAverage}%
                  </p>
                </div>
              </div>
            </CardContent>
          </Card>

          {/* Valuation Breakdown */}
          <Card>
            <CardHeader>
              <CardTitle>{t("valuationBreakdown")}</CardTitle>
            </CardHeader>
            <CardContent>
              <div className="space-y-4">
                <div className="flex justify-between items-center py-2 border-b">
                  <span>{t("baseValue")}</span>
                  <span className="font-medium">{formatCurrency(result.baseValue)}</span>
                </div>
                <div className="flex justify-between items-center py-2 border-b">
                  <span>{t("ageAdjustment")}</span>
                  <span
                    className={`font-medium ${
                      result.ageAdjustment >= 0 ? "text-green-600" : "text-red-600"
                    }`}
                  >
                    {result.ageAdjustment >= 0 ? "+" : ""}
                    {formatCurrency(result.ageAdjustment)}
                  </span>
                </div>
                <div className="flex justify-between items-center py-2 border-b">
                  <span>{t("conditionAdjustment")}</span>
                  <span
                    className={`font-medium ${
                      result.conditionAdjustment >= 0 ? "text-green-600" : "text-red-600"
                    }`}
                  >
                    {result.conditionAdjustment >= 0 ? "+" : ""}
                    {formatCurrency(result.conditionAdjustment)}
                  </span>
                </div>
                {result.featureBonus > 0 && (
                  <div className="flex justify-between items-center py-2 border-b">
                    <span>{t("featureBonus")}</span>
                    <span className="font-medium text-green-600">
                      +{formatCurrency(result.featureBonus)}
                    </span>
                  </div>
                )}
                <div className="flex justify-between items-center py-2 bg-muted px-4 rounded-lg">
                  <span className="font-medium">{t("totalAdjustments")}</span>
                  <span
                    className={`font-bold ${
                      result.totalAdjustments >= 0 ? "text-green-600" : "text-red-600"
                    }`}
                  >
                    {result.totalAdjustments >= 0 ? "+" : ""}
                    {formatCurrency(result.totalAdjustments)}
                  </span>
                </div>
              </div>
            </CardContent>
          </Card>

          {/* Selected Features */}
          {features.length > 0 && (
            <Card>
              <CardHeader>
                <CardTitle>{t("selectedFeatures")}</CardTitle>
              </CardHeader>
              <CardContent>
                <div className="flex flex-wrap gap-2">
                  {features.map((featureId) => {
                    const feature = PROPERTY_FEATURES.find((f) => f.id === featureId);
                    return (
                      <Badge key={featureId} variant="secondary">
                        {t(`featureNames.${featureId}`)}
                        {feature && ` (+${formatCurrency(feature.bonus)})`}
                      </Badge>
                    );
                  })}
                </div>
              </CardContent>
            </Card>
          )}
        </>
      )}

      {/* Reset Button */}
      <div className="flex justify-end">
        <Button variant="outline" onClick={reset}>
          <RotateCcw className="h-4 w-4 mr-2" />
          {commonT("reset")}
        </Button>
      </div>
    </div>
  );
}
