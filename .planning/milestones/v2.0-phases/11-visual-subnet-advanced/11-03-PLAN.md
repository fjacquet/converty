**Update subnet-calculator.tsx:**

Add mode tabs and integrate new components.

1. Import new components:

```typescript
import { SplitControls } from "./components/split-controls";
import { SubnetTree } from "./components/subnet-tree";
import { SupernetInput } from "./components/supernet-input";
import { ComparisonPanel } from "./components/comparison-panel";
import { Tabs, TabsContent, TabsList, TabsTrigger } from "@/components/ui/tabs";
```

1. Get new state from store:

```typescript
const {
  // existing
  ipInput, subnetMask, result, error, setIPInput, setSubnetMask, calculate, reset,
  // new
  mode, setMode, divisionCount, setDivisionCount, performDivision, subnetDivision,
  networksInput, setNetworksInput, performAggregation, supernetResult,
} = useSubnetCalculatorStore();
```

1. Add mode tabs below the basic calculator inputs:

```tsx
<Tabs value={mode} onValueChange={(v) => setMode(v as CalculatorMode)}>
  <TabsList className="grid w-full grid-cols-3">
    <TabsTrigger value="basic">{t("mode-basic")}</TabsTrigger>
    <TabsTrigger value="subnetting">{t("mode-subnetting")}</TabsTrigger>
    <TabsTrigger value="supernetting">{t("mode-supernetting")}</TabsTrigger>
  </TabsList>

  <TabsContent value="basic">
    {/* Existing basic calculator content */}
  </TabsContent>

  <TabsContent value="subnetting">
    {result && (
      <>
        <SplitControls
          currentCidr={result.cidr}
          ipVersion={result.ipVersion}
          divisionCount={divisionCount}
          onDivisionCountChange={setDivisionCount}
          onSplit={performDivision}
          disabled={!result}
        />
        {subnetDivision && (
          <>
            <SubnetTree division={subnetDivision} />
            <ComparisonPanel
              mode="subnetting"
              before={subnetDivision.parent}
              after={subnetDivision.children}
            />
          </>
        )}
      </>
    )}
    {!result && <p>{t("calculate-first")}</p>}
  </TabsContent>

  <TabsContent value="supernetting">
    <SupernetInput
      value={networksInput}
      onChange={setNetworksInput}
      onAggregate={performAggregation}
      error={error}
    />
    {supernetResult?.success && (
      <ComparisonPanel
        mode="supernetting"
        before={supernetResult.originalNetworks}
        after={supernetResult.supernet}
      />
    )}
  </TabsContent>
</Tabs>
```

1. Keep existing visualizations (network diagram, binary representation, breakdown table) visible in basic mode and when result exists.
