The page should:
1. Import necessary components and hooks
2. Use getTranslations for server-side translations
3. Display category name and description
4. List all calculators in the infrastructure category (currently empty)
5. Use existing dynamic import pattern for performance (leverages Phase 21 lazy loading infrastructure)

**Pattern to follow (simplified):**
```typescript
import { getTranslations } from "next-intl/server";
import { Metadata } from "next";
import { CategoryPageLayout } from "@/components/category-page-layout";
import { getConvertersByCategory } from "@/lib/registry/converters";
import { getCategoryBySlug } from "@/lib/registry/categories";

export async function generateMetadata({ params }: { params: Promise<{ locale: string }> }): Promise<Metadata> {
  const { locale } = await params;
  const t = await getTranslations({ locale, namespace: "categories" });

  return {
    title: t("infrastructure.name"),
    description: t("infrastructure.description"),
  };
}

export default async function InfrastructurePage({ params }: { params: Promise<{ locale: string }> }) {
  const { locale } = await params;
  const t = await getTranslations({ locale });
  const category = getCategoryBySlug("infrastructure");
  const converters = getConvertersByCategory("infrastructure");

  if (!category) {
    return null;
  }

  return (
    <CategoryPageLayout
      category={category}
      converters={converters}
      locale={locale}
      t={t}
    />
  );
}
```

**Notes:**
- Check actual network/page.tsx for exact component imports and patterns

- Use async params pattern (Next.js 16)
