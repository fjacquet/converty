```bash
node -e "
const en = require('./src/messages/en.json');
const fr = require('./src/messages/fr.json');
const de = require('./src/messages/de.json');
const it = require('./src/messages/it.json');

function flatKeys(obj, prefix = '') {
  return Object.entries(obj).flatMap(([k, v]) =>
    typeof v === 'object' ? flatKeys(v, prefix ? prefix + '.' + k : k) : [prefix ? prefix + '.' + k : k]
  );
}

const ns = 'server-refresh-calculator';
const enKeys = flatKeys(en.converters[ns]);
for (const [lang, msg] of [['fr', fr], ['de', de], ['it', it]]) {
  const langKeys = flatKeys(msg.converters[ns]);
  const missing = enKeys.filter(k => !langKeys.includes(k));
  if (missing.length > 0) {
    console.log(lang.toUpperCase() + ' MISSING:', missing.join(', '));
  } else {
    console.log(lang.toUpperCase() + ': ALL KEYS PRESENT');
  }
}
"
```
